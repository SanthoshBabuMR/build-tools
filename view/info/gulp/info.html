About Grunt
  • Gulp is a Stream based build system
  • Runs on top of Node.js
  • Grunt setup based on code over configuration
  • Requires package.json and gulpfile.js to be configured at the project's root directory
  • package.json contains project's meta data such as project name, description, author, license. It also contains the list of depenedencies needed by the project
  • gulpfile.js is used for requiring gulp plugins and defining tasks
  • Gulp is all about streams and building complex pipelines with ease. It uses node.jsâ€™ streams, and executes faster, since it does not open/close files, or create intermediary copies all the time.
  • APIs: gulp.src, gulp.desc, gulp.watch
Sample package.json
{
  "name": "name-of-the-application",
  "version": "0.0.0",
  "description": "description/purpose of the application",
  "main": "index.js",
  "scripts": {
    "test": "echo \"Error: no test specified\" && exit 1"
  },
  "repository" : { 
    "type" : "git|svn",
    "url"  : "path-to-repo"
  },
  "author": "name-of-the-author",
  "license": "ISC",
  "devDependencies": {
    "gulp": "^3.8.11",
    "gulp-connect": "^2.2.0"
  }
}
Sample gulpfile.js
// require gulp plugins
var gulp    = require('gulp'),
    connect = require('gulp-connect');

// configure task
gulp.task('connect', function() {
  connect.server({
    port: 62342
  });
});
 
// register task
gulp.task('startServer', ['connect']);
