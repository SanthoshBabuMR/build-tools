About Grunt
  • Grunt is a JavaScript Task Runner
  • Runs on top of Node.js
  • Grunt setup based on configuration rather than code
  • Requires package.json and Gruntfile.js to be configured at the project's root directory
  • package.json contains project's meta data such as project name, description, author, license. It also contains the list of depenedencies needed by the project
  • Gruntfile.js is used for defining tasks and loading grunt plugins used in configuring the task.
  • Advantages include matured ecosystem with large number of plugins and an active development community.
    The barrier to entry is relatively low because a vast range of tools and automated tasks are already available to use.
Sample package.json
                              
{
  "name": "name-of-the-application",
  "version": "0.0.0",
  "description": "description/purpose of the application",
  "main": "index.js",
  "scripts": {
    "test": "echo \"Error: no test specified\" && exit 1"
  },
  "repository" : { 
    "type" : "git|svn",
    "url"  : "path-to-repo"
  },
  "author": "name-of-the-author",
  "license": "ISC",
  "devDependencies": {
    "grunt": "^0.4.5",
    "grunt-contrib-connect": "^0.10.1"
  }
}
Sample Gruntfile.js

Configuration over code 

                              
module.exports = function( grunt ) {
  // configure task(s)
  grunt.initConfig( {
    connect: {
      appServer: {
        options: {
          protocol: "https"
          , hostname: "localhost"
          , port: 6234
          , base: "."
          , keepalive: true
        }
      }
    }    
  } );
  // load task(s)
  grunt.loadNpmTasks( "grunt-contrib-connect" );
  // register task(s)
  grunt.registerTask( "appServer", "connect:appServer" );
};
